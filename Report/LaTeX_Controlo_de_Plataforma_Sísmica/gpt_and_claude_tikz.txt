\begin{tikzpicture}[
  block/.style    = {draw, rectangle, minimum width=2.2cm, minimum height=0.9cm,
                     align=center, font=\itshape},
  bigblock/.style = {draw, rectangle, minimum width=2.8cm, minimum height=1.8cm,
                     align=center, font=\itshape},
  circ/.style     = {draw, circle, minimum size=7mm, inner sep=0pt},
  >=Stealth, thick
]

%--- Nodes ---
\node[circ]     (sum1)  at (0,    0  ) {$\times$};
\node[block]    (C)     at (2.8,  0  ) {$C(z^{-1})$};
\node[block]    (G)     at (6.2,  0  ) {$G(z^{-1})$};
\node[block]    (Ghat)  at (6.2,  2.5) {$\hat{G}(z^{-1})$};
\node[block]    (H)     at (6.2, -2.5) {$H(z^{-1})$};
\node[circ]     (sum2)  at (9.5,  2.5) {$\times$};
\node[bigblock] (ident) at (9.5,  5.8) {Identification\\algorithm};
\node[bigblock] (ctrl)  at (3.5,  5.8) {Control\\design\\algorithm};

%--- Junction coordinates ---
\coordinate (ubranch) at (4.7, 0  );   % u splits here
\coordinate (ybranch) at (7.8, 0  );   % y splits here
\coordinate (ysum)    at (9.5, 0  );   % below sum2

%--- Input r -> sum1 ---
\draw[->] (-1.5, 0) node[left]{$r$} -- (sum1);

%--- Main forward path: sum1 -> C -> G ---
\draw[->] (sum1)   -- (C);
\draw     (C)      -- node[above]{$u$} (ubranch);
\draw[->] (ubranch)-- (G);
\filldraw (ubranch) circle (2pt);

%--- G output -> y, with junctions ---
\draw     (G)       -- node[above]{$y$} (ybranch);
\draw     (ybranch) -- (ysum);
\draw[->] (ysum)    -- ++(0.8, 0);          % y continues as system output
\filldraw (ybranch) circle (2pt);
\filldraw (ysum)    circle (2pt);

%--- u branch up -> Ghat (vertical then horizontal) ---
\draw[->] (ubranch) |- (Ghat.west);

%--- Ghat -> yhat -> sum2 ---
\draw[->] (Ghat.east) -- node[above]{$\hat{y}$} (sum2.west);

%--- y enters sum2 from below ---
\draw[->] (ysum) -- (sum2.south);

%--- sum2 +/- sign labels ---
\node[font=\small, above right] at (sum2.west)  {$-$};
\node[font=\small, above left]  at (sum2.south) {$+$};

%--- sum2 -> identification (offset right to avoid overlap) ---
\draw[->] ([xshift= 2mm]sum2.north) -- ([xshift= 2mm]ident.south);

%--- identification -> Ghat parameter update (offset left) ---
\draw[->] ([xshift=-2mm]ident.south) -- ++(0,-0.9) -| (Ghat.north);

%--- identification -> control design ---
\draw[->] (ident.west) -- (ctrl.east);

%--- control design -> C (diagonal) ---
\draw[->] (ctrl.south) -- (C.north);

%--- y -> H (branch downward) ---
\draw     (ybranch) -- ++(0,-2.5) -- (H.east);

%--- H -> tilde_y -> sum1 (bottom feedback) ---
\draw     (H.west)    -- node[above]{$\tilde{y}$} (-0.5,-2.5);
\draw[->] (-0.5,-2.5) |- (sum1);

\end{tikzpicture}
