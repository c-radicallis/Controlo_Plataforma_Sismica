        m_T \ddot{x}_T &+ c_1 (\dot{x}_T - \dot{x}_1) + k_1 (x_T - x_1) = F_p \\
        m_1 \ddot{x}_1 &+ c_1 (\dot{x}_1 - \dot{x}_T) + c_2 (\dot{x}_1 - \dot{x}_2) + k_1 (x_1 - x_T) + k_2 (x_1 - x_2) = 0 \\
        m_2 \ddot{x}_2 &+ c_2 (\dot{x}_2 - \dot{x}_1) + k_2 (x_2 - x_1) = 0
    \end{aligned}
    \right. 
    \Longleftrightarrow
    \end{align}

    \end{equation*}
