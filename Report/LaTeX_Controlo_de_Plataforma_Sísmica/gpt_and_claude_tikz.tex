\begin{tikzpicture}[
  block/.style    = {draw, rectangle, minimum width=2.2cm, minimum height=0.9cm,
                     align=center, font=\itshape},
  bigblock/.style = {draw, rectangle, minimum width=2.8cm, minimum height=1.8cm,
                     align=center, font=\itshape},
  circ/.style     = {draw, circle, minimum size=7mm, inner sep=0pt},
  >=Stealth, thick

%--- Nodes ---
\node[circ]     (sum1)  at (0,    0  ) {$\times$};
\node[block]    (C)     at (2.8,  0  ) {$C(z^{-1})$};
\node[block]    (G)     at (6.2,  0  ) {$G(z^{-1})$};

\node[bigblock] (ctrl)  at (3.5,  5.8) {Control\\design\\algorithm};

\coordinate (ybranch) at (7.8, 0  );   % y splits here
\coordinate (ysum)    at (9.5, 0  );   % below sum2

%--- Main forward path: sum1 -> C -> G ---

%--- y enters sum2 from below ---
\draw[->] (ysum) -- (sum2.south);

%--- control design -> C (diagonal) ---
